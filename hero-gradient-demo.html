üìù CSS Code Preview

/* CSS Variables */
--hero-gradient-navy: linear-gradient(135deg, #0c1a3a 0%, #0A0F1E 50%, #0c1a3a 100%);
--hero-overlay-gold: rgba(212, 175, 55, 0.1);

/* Hero Gradient Background */
.hero, .hero-section {
  position: relative;
  background: var(--hero-gradient-navy);
  overflow: hidden;
}

/* Gold Overlay */
.hero::before, .hero-section::before {
  content: '';
  position: absolute;
  inset: 0;
  background: var(--hero-overlay-gold);
  pointer-events: none;
  z-index: 0;
}
